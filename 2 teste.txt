        let selectedService = {};
        let selectedDate = null;
        let selectedTime = null;
        let customerData = {};
        let whatsappLink = '';
        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            document.getElementById(screenId).classList.add('active');
        }
        function selectService(id, name, price, duration, icon) {
            selectedService = { id, name, price, duration, icon };
            document.getElementById('selected-service-icon').textContent = icon;
            document.getElementById('selected-service-name').textContent = name;
            document.getElementById('selected-service-duration').textContent = duration + ' ‚Ä¢ R$ ' + price;
            document.getElementById('selected-service-price').textContent = 'R$ ' + price;
            showScreen('datetime');
            generateCalendar();
        }
        function generateCalendar() {
            const calendar = document.getElementById('calendar');
            calendar.innerHTML = '';
            const today = new Date();
            for (let i = 0; i < 7; i++) {
                const date = new Date(today);
                date.setDate(today.getDate() + i);
                const dayBtn = document.createElement('div');
                dayBtn.className = 'day-btn';
                dayBtn.onclick = () => selectDate(date, dayBtn);
                dayBtn.innerHTML = `<div class="day-weekday">${date.toLocaleDateString('pt-BR', { weekday: 'short' })}</div><div class="day-number">${date.getDate()}</div><div class="day-month">${date.toLocaleDateString('pt-BR', { month: 'short' })}</div>`;
                calendar.appendChild(dayBtn);
            }
        }
        function selectDate(date, btn) {
            document.querySelectorAll('.day-btn').forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            selectedDate = date;
            generateTimeSlots();
            document.getElementById('time-section').style.display = 'block';
        }
        function generateTimeSlots() {
            const timegrid = document.getElementById('timegrid');
            timegrid.innerHTML = '';
            const times = ['09:00', '09:30', '10:00', '10:30', '11:00', '11:30', '14:00', '14:30', '15:00', '15:30', '16:00', '16:30', '17:00', '17:30', '18:00', '18:30', '19:00', '19:30'];
            times.forEach(time => {
                const timeBtn = document.createElement('div');
                timeBtn.className = 'time-btn';
                timeBtn.textContent = time;
                timeBtn.onclick = () => selectTime(time, timeBtn);
                timegrid.appendChild(timeBtn);
            });
        }
        function selectTime(time, btn) {
            document.querySelectorAll('.time-btn').forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            selectedTime = time;
            document.getElementById('continue-btn').style.display = 'block';
        }
        function showScreen(screen) {
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            document.getElementById(screen).classList.add('active');
            if (screen === 'customer') {
                document.getElementById('summary-service').textContent = selectedService.name;
                document.getElementById('summary-date').textContent = selectedDate.toLocaleDateString('pt-BR');
                document.getElementById('summary-time').textContent = selectedTime;
                document.getElementById('summary-price').textContent = 'R$ ' + selectedService.price;
            }
        }
        function confirmBooking() {
            const name = document.getElementById('customer-name').value;
            const phone = document.getElementById('customer-phone').value;
            const email = document.getElementById('customer-email').value;
            if (!name || !phone) {
                alert('Por favor, preencha todos os campos obrigat√≥rios');
                return;
            }
            customerData = { name, phone, email };
            document.getElementById('confirm-name').textContent = name;
            document.getElementById('confirm-service').textContent = selectedService.name;
            document.getElementById('confirm-date').textContent = selectedDate.toLocaleDateString('pt-BR');
            document.getElementById('confirm-time').textContent = selectedTime;
            document.getElementById('confirm-phone').textContent = phone;
            const message = `üî• *NOVO AGENDAMENTO - MATHEUS BARBER SHOP* üî•\n\nüë§ *Cliente:* ${name}\nüìû *Telefone:* ${phone}\n${email ? `üìß *Email:* ${email}\n` : ''}\nüíà *Servi√ßo:* ${selectedService.name}\nüìÖ *Data:* ${selectedDate.toLocaleDateString('pt-BR')}\nüïê *Hor√°rio:* ${selectedTime}\nüí∞ *Valor:* R$ ${selectedService.price}\n\n‚úÖ Agendamento confirmado!`;
            whatsappLink = `https://wa.me/5534991054546?text=${encodeURIComponent(message)}`;
            showScreen('confirmation');
            setTimeout(() => window.open(whatsappLink, '_blank'), 1000);
        }
        function resetBooking() {
            selectedService = {};
            selectedDate = null;
            selectedTime = null;
            customerData = {};
            document.getElementById('customer-name').value = '';
            document.getElementById('customer-phone').value = '';
            document.getElementById('customer-email').value = '';
            showScreen('home');
        }
